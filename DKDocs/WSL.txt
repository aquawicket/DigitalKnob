###	Fix WSL retaining file permissions ### https://superuser.com/a/1392722/600216

	/etc/wsl.conf

	[boot]
	systemd=true

	[automount]
	enabled  = true
	root     = /mnt/
	options  = "metadata,umask=22,fmask=11"
